/**
 * Get human-readable label for a section ID.
 */
export function formatSectionName(sectionId: string): string {
  return SECTION_LABELS[sectionId] || sectionId.replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase());
}
